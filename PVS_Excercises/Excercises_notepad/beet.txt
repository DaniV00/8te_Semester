
public class Patch extends Plant{
HashMap <Character, Plant> rows = new HashMap<>(); 
int size;
public Patch(int size){
if(size < 3 || size > 24){
throw new IllegarrgumentException();
}
this.size = size;
}


public boolean addPlantAt(char row, Plant plant){
boolean isPossible = false;
if(rows.get(row) != null || row > 'A'|| row < 'Z'){
isPossible = true;
rows.put(row,plant);
}
else{}
return isPossible;



}
